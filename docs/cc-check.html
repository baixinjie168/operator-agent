0 概述

在 LangGraph 管线的约束检查 Node 中，创建隔离的 Claude Code 工作空间，从 .env 读取模型配置，通过 subprocess 调用 claude -p 完成约束检查。

目标 将 constraint_check_agent.py 从 DeepAgents + LocalShellBackend 改为 Claude Code CLI。
配置全部来自 .env（通过 Settings 类），不依赖全局 ~/.claude/settings.json。
调用方（routes/query.py、routes/task.py、task_engine.py）接口不变。

# packages/agent/src/agent/core/config.py -- Settings 类新增

class Settings(BaseSettings):
    # ... 已有字段不变 ...

    # --- Claude Code CLI for constraint check ---
    constraint_check_cc_enabled: bool = False
    constraint_check_cc_base_url: str = ""
    constraint_check_cc_timeout: int = Field(default=600, ge=60, le=3600)
    constraint_check_cc_cli_path: str = "claude"

    def get_cc_settings(self) -> dict:
        """Generate workspace/.claude/settings.json content from .env."""
        provider = self.constraint_check_llm_provider
        api_key = self.get_provider_api_key(provider)
        model = self.constraint_check_model or self.get_provider_model(provider)
        base_url = self.constraint_check_cc_base_url

        if not api_key:
            raise ValueError(
                f"API key for provider '{provider}' is required. "
                f"Set {provider.value.upper()}_API_KEY in .env."
            )
        if not base_url:
            raise ValueError(
                "CONSTRAINT_CHECK_CC_BASE_URL is required when "
                "CONSTRAINT_CHECK_CC_ENABLED=true"
            )

        return {
            "env": {
                "ANTHROPIC_AUTH_TOKEN": api_key,
                "ANTHROPIC_BASE_URL": base_url,
                "ANTHROPIC_DEFAULT_SONNET_MODEL": model,
                "ANTHROPIC_DEFAULT_OPUS_MODEL": model,
            },
            "permissions": {
                "allow": ["Read(*)", "Write(*)"],
                "deny": [],
            },
        }

4 完整实现代码

4.1 constraint_check_agent.py（完整重写）

# packages/agent/src/agent/nodes/constraint_check_agent.py
"""Constraint check via Claude Code CLI.

Creates an isolated workspace for each check, configures the model
from .env via Settings, invokes `claude -p` as a subprocess.
No dependency on DeepAgents, ChatOpenAI, or global ~/.claude/settings.json.
"""

from __future__ import annotations

import asyncio
import json
import logging
import os
import shutil
import subprocess
import tempfile

from agent.core.config import settings

logger = logging.getLogger(__name__)

_PROJECT_ROOT = os.path.abspath(
    os.path.join(os.path.dirname(__file__), "..", "..", "..", "..", "..")
)
_SKILL_SOURCE = os.path.join(
    _PROJECT_ROOT, ".claude", "skills", "operator-constraint-checker"
)

_check_lock = asyncio.Lock()


def _create_workspace(operator_name, markdown_content, json_constraints):
    """Create isolated Claude Code workspace."""
    workspace = tempfile.mkdtemp(prefix=f"claude-check-{operator_name}-")

    # 1a. Copy skill
    skill_dest = os.path.join(
        workspace, ".claude", "skills", "operator-constraint-checker"
    )
    shutil.copytree(_SKILL_SOURCE, skill_dest, dirs_exist_ok=True)

    # 1b. Write model config (from .env via Settings)
    cc_settings = settings.get_cc_settings()
    os.makedirs(os.path.join(workspace, ".claude"), exist_ok=True)
    with open(os.path.join(workspace, ".claude", "settings.json"), "w") as f:
        json.dump(cc_settings, f, indent=2)

    # 1c. Write input files
    with open(os.path.join(workspace, "operator.md"), "w", encoding="utf-8") as f:
        f.write(markdown_content)
    with open(os.path.join(workspace, "constraints.json"), "w", encoding="utf-8") as f:
        f.write(json_constraints)

    logger.info("ClaudeCLI: workspace created at %s", workspace)
    return workspace


_PROMPT = (
    "使用 operator-constraint-checker skill 检查约束。"
    "算子文档：operator.md，"
    "约束文档：constraints.json，"
    "输出路径：report.html"
)


async def run_constraint_check(markdown_content, json_constraints, operator_name):
    """Run constraint check via Claude Code CLI.

    Interface unchanged -- callers need no modification.
    Returns HTML report string, or "" on failure.
    """
    if not markdown_content.strip() or not json_constraints.strip():
        return ""

    async with _check_lock:
        return await _run_cli_check(markdown_content, json_constraints, operator_name)


async def _run_cli_check(markdown_content, json_constraints, operator_name):
    """Actual CLI invocation (called under _check_lock)."""
    workspace = _create_workspace(operator_name, markdown_content, json_constraints)

    cli_path = settings.constraint_check_cc_cli_path or "claude"
    cmd = [
        cli_path, "-p", _PROMPT,
        "--model", "sonnet",
        "--allowedTools", "Read Write",
        "--output-format", "json",
    ]

    timeout = settings.constraint_check_cc_timeout
    report_path = os.path.join(workspace, "report.html")

    logger.info(
        "ClaudeCLI: starting check for %s (cwd=%s, timeout=%ds)",
        operator_name, workspace, timeout,
    )

    try:
        result = await asyncio.to_thread(
            subprocess.run, cmd,
            cwd=workspace, capture_output=True, text=True, timeout=timeout,
        )

        if result.returncode != 0:
            logger.error("ClaudeCLI: %s exited %d, stderr: %s",
                operator_name, result.returncode,
                (result.stderr or "")[:500])
        else:
            logger.info("ClaudeCLI: %s completed (stdout=%d chars)",
                operator_name, len(result.stdout))

        # Read HTML report
        if os.path.exists(report_path):
            with open(report_path, "r", encoding="utf-8") as f:
                html = f.read()
            logger.info("ClaudeCLI: report read for %s (%d chars)",
                operator_name, len(html))
            return html

        # Fallback: extract from JSON stdout
        if result.stdout:
            try:
                data = json.loads(result.stdout)
                content = data.get("result", "")
                if "<html" in content.lower():
                    return content
            except json.JSONDecodeError:
                pass

        logger.error("ClaudeCLI: no report found for %s", operator_name)
        return ""

    except subprocess.TimeoutExpired:
        logger.error("ClaudeCLI: timeout for %s after %ds", operator_name, timeout)
        return ""
    except FileNotFoundError:
        logger.error("ClaudeCLI: '%s' not found. Set CONSTRAINT_CHECK_CC_CLI_PATH", cli_path)
        return ""
    except Exception as e:
        logger.exception("ClaudeCLI: failed for %s: %s", operator_name, e)
        return ""
    finally:
        shutil.rmtree(workspace, ignore_errors=True)
        logger.info("ClaudeCLI: workspace cleaned for %s", operator_name)

5 隔离性分析

维度隔离方式说明
文件系统cwd=workspace (tempdir)Claude 只能 Read/Write workspace 内文件。项目代码、全局配置不可见。
模型配置workspace/.claude/settings.json项目级 settings 覆盖全局。API Key / Base URL / Model 全部从 .env 读取。
Skill复制到 workspace/.claude/skills/从项目 .claude/skills/operator-constraint-checker/ 复制。不依赖全局 skill。
工具权限--allowedTools "Read Write"禁止 Bash/Grep/Glob，只允许读写文件。
进程隔离subprocessClaude 是独立进程，崩溃不影响 FastAPI 主进程。
并发隔离_check_lock + subprocess 串行同时只有一个 Claude 实例运行。
生命周期finally: shutil.rmtree()工作空间用完即删，不留痕迹。
注意：全局 settings 仍会被加载 Claude CLI 非 --bare 模式下，全局 ~/.claude/settings.json 仍会被读取。但项目级 workspace/.claude/settings.json 的 env 字段会覆盖全局同名变量。如果需要完全隔离，添加 --bare 标志（但需改用 ANTHROPIC_API_KEY 而非 ANTHROPIC_AUTH_TOKEN）。

7 Fallback 策略（可选）

当 CONSTRAINT_CHECK_CC_ENABLED=false 或 CLI 不可用时，自动 fallback 到 DeepAgents：

# constraint_check_agent.py -- run_constraint_check 入口

async def run_constraint_check(markdown_content, json_constraints, operator_name):
    if not markdown_content.strip() or not json_constraints.strip():
        return ""

    async with _check_lock:
        if settings.constraint_check_cc_enabled:
            # 方式 A: Claude Code CLI
            html = await _run_cli_check(markdown_content, json_constraints, operator_name)
            if html:
                return html
            logger.warning("ClaudeCLI failed, falling back to DeepAgents")

        # 方式 B: DeepAgents fallback
        return await _run_deepagents_check(markdown_content, json_constraints, operator_name)

    from agent.nodes._constraint_check_da import run_constraint_check as _da_check

关于 --model sonnet

Claude CLI 的 --model 接受别名（sonnet/opus/haiku），不直接写模型名。别名映射到 settings.json 中的环境变量：

--model sonnet -> env.ANTHROPIC_DEFAULT_SONNET_MODEL -> glm-5.2
--model opus   -> env.ANTHROPIC_DEFAULT_OPUS_MODEL   -> glm-5.2
--model haiku  -> env.ANTHROPIC_DEFAULT_HAIKU_MODEL  -> (未设置)

在 get_cc_settings() 中，我们将 SONNET 和 OPUS 都设为 constraint_check_model，所以 --model sonnet 就会用 .env 中的模型。
